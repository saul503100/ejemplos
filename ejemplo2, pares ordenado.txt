using System;
using System.Linq;

class WhereSample2
{
	static void Main()
	{
		int[] scores = { 97, 92, 81, 60 };
		
		var queryLowNums =
			from num in scores
			where IsEven(num)
			select num;
		
		foreach (var s in queryLowNums)
		{
			Console.Write(s.ToString()+ " ");		
		}
	}
	static bool IsEven(int i)
	{
		return i %2 ==0;
	}
}
